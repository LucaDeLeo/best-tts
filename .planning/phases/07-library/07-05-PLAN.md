5. **Update getLibraryItems in library-storage.ts** to support filtering:

```typescript
export async function getLibraryItems(options?: {
  folderId?: string | null;
  sortBy?: 'lastReadAt' | 'createdAt';
}): Promise<LibraryItem[]> {
  const db = await getLibraryDB();
  let items: LibraryItem[];

  if (options?.folderId !== undefined) {
    // Filter by folder (null = root)
    const index = db.transaction('library-items').store.index('by-folderId');
    items = await index.getAll(options.folderId);
  } else {
    // All items
    items = await db.getAll('library-items');
  }

  // Sort by lastReadAt descending
  items.sort((a, b) => b.lastReadAt - a.lastReadAt);
  return items;
}
```

```typescript
import React, { useState, useEffect } from 'react';
import { MessageType } from '../lib/messages';
import type { LibraryItem } from '../lib/library-types';

interface RecentItemsProps {
  onPlayItem: (itemId: string) => void;
  onOpenLibrary: () => void;
}

export function RecentItems({ onPlayItem, onOpenLibrary }: RecentItemsProps) {
  const [items, setItems] = useState<LibraryItem[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadRecentItems();
  }, []);

  const loadRecentItems = async () => {
    setLoading(true);
    const response = await chrome.runtime.sendMessage({
      target: 'service-worker',
      type: MessageType.GET_RECENT_ITEMS,
      limit: 5
    });
    if (response.success) {
      setItems(response.items);
    }
    setLoading(false);
  };

  const calculateProgress = (item: LibraryItem): number | null => {
    if (!item.resumeData) return null;
    const { currentChunkIndex, contentLength, charOffset } = item.resumeData;
    if (contentLength && charOffset) {
      return Math.round((charOffset / contentLength) * 100);
    }
    return null;
  };

  if (loading) {
    return <div className="recent-items loading">Loading...</div>;
  }

  if (items.length === 0) {
    return (
      <div className="recent-items empty">
        <p>No saved items yet</p>
        <p className="hint">Save a page or document to see it here</p>
      </div>
    );
  }

  return (
    <div className="recent-items">
      <div className="recent-header">
        <h3>Recent</h3>
        <button className="view-all-btn" onClick={onOpenLibrary}>
          View All
        </button>
      </div>
      <ul className="recent-list">
        {items.map(item => {
          const progress = calculateProgress(item);
          return (
            <li
              key={item.id}
              className={`recent-item ${item.contentDeleted ? 'content-deleted' : ''}`}
              onClick={() => !item.contentDeleted && onPlayItem(item.id)}
            >
              <div className="item-content">
                <span className="item-title">{item.title}</span>
                <span className="item-source">{item.source}</span>
              </div>
              {progress !== null && (
                <div className="progress-indicator">
                  <div
                    className="progress-bar"
                    style={{ width: `${progress}%` }}
                  />
                  <span className="progress-text">{progress}%</span>
                </div>
              )}
              {item.contentDeleted && (
                <span className="deleted-badge" title="Content removed to save space">
                  Needs re-extract
                </span>
              )}
            </li>
          );
        })}
      </ul>
    </div>
  );
}
```
