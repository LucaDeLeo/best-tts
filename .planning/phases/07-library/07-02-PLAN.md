4. **Add SAVE_TO_LIBRARY handler** in handleServiceWorkerMessage:
```typescript
case MessageType.SAVE_TO_LIBRARY: {
  const { item } = message as SaveToLibraryMessage;

  // Check quota
  const quota = await getStorageEstimateForLibrary();
  const contentSize = new Blob([item.content]).size;
  if (quota && quota.available < contentSize + 5 * 1024 * 1024) {
    sendResponse({ success: false, error: 'Storage full' });
    break;
  }

  // Check if already saved
  const existing = await isUrlSaved(item.url);
  if (existing) {
    sendResponse({ success: true, alreadyExists: true, itemId: existing.id });
    break;
  }

  // Create and save
  const libraryItem: LibraryItem = {
    id: crypto.randomUUID(),
    title: item.title,
    url: item.url,
    source: item.source,
    folderId: item.folderId || null,
    createdAt: Date.now(),
    lastReadAt: Date.now(),
    contentDeleted: false,
    contentDeletedAt: null,
    contentSize: contentSize,
    contentHash: await hashContent(item.content),
    resumeData: null
  };

  await saveLibraryItem(libraryItem, item.content);
  sendResponse({ success: true, itemId: libraryItem.id });
  break;
}
```

```typescript
const handleSaveToLibrary = async () => {
  if (!extractedText || savingToLibrary) return;

  setSavingToLibrary(true);
  try {
    const response = await chrome.runtime.sendMessage({
      target: 'service-worker',
      type: MessageType.SAVE_TO_LIBRARY,
      item: {
        title: documentTitle || 'Untitled',
        url: documentUrl || '',
        source: 'webpage', // or derive from source
        content: extractedText
      }
    });

    if (response.success) {
      setIsSavedToLibrary(true);
      if (response.alreadyExists) {
        // Already in library, could show a toast
      }
    } else {
      console.error('Save failed:', response.error);
    }
  } finally {
    setSavingToLibrary(false);
  }
};
```
