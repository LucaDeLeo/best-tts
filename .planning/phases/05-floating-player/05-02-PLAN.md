<task type="auto">
  <name>Task 2: Add keyboard shortcuts for focused player</name>
  <files>src/content/floating-player.ts</files>
  <action>
Add focus-scoped keyboard handlers per CONTEXT.md decision [12]:

1. **Add keydown listener to player container:**
   ```typescript
   container.addEventListener('keydown', (e) => {
     // Only handle when player container has focus
     switch (e.key) {
       case ' ':
       case 'Enter':
         e.preventDefault();
         // Toggle play/pause
         if (currentState.status === 'playing') {
           sendPlaybackCommand(MessageType.PAUSE_AUDIO);
         } else if (currentState.status === 'paused') {
           sendPlaybackCommand(MessageType.RESUME_AUDIO);
         }
         break;

       case 'Escape':
         e.preventDefault();
         // Dismiss player
         sendPlaybackCommand(MessageType.STOP_PLAYBACK);
         break;

       case 'ArrowLeft':
         e.preventDefault();
         // Skip previous
         const prevIndex = Math.max(0, currentState.currentChunk - 1);
         if (prevIndex !== currentState.currentChunk) {
           sendPlaybackCommand(MessageType.SKIP_TO_CHUNK, { chunkIndex: prevIndex });
         }
         break;

       case 'ArrowRight':
         e.preventDefault();
         // Skip next
         const nextIndex = Math.min(currentState.totalChunks - 1, currentState.currentChunk + 1);
         if (nextIndex !== currentState.currentChunk) {
           sendPlaybackCommand(MessageType.SKIP_TO_CHUNK, { chunkIndex: nextIndex });
         }
         break;

       case 'ArrowUp':
         e.preventDefault();
         // Increase speed
         const fasterIdx = SPEED_PRESETS.findIndex(s => s > currentState.playbackSpeed);
         if (fasterIdx !== -1) {
           sendPlaybackCommand(MessageType.SET_SPEED, { speed: SPEED_PRESETS[fasterIdx] });
         }
         break;

       case 'ArrowDown':
         e.preventDefault();
         // Decrease speed
         const speeds = [...SPEED_PRESETS].reverse();
         const slowerIdx = speeds.findIndex(s => s < currentState.playbackSpeed);
         if (slowerIdx !== -1) {
           sendPlaybackCommand(MessageType.SET_SPEED, { speed: speeds[slowerIdx] });
         }
         break;
     }
   });
   ```
