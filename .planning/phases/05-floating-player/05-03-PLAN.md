   ```typescript
   async function handleStatusUpdate(message: any): Promise<{ success: boolean }> {
     const status = message.status;

     // Map service worker status to PlayerUIState
     let playerStatus: 'idle' | 'generating' | 'playing' | 'paused';
     if (status.isPlaying) {
       playerStatus = 'playing';
     } else if (status.isPaused) {
       playerStatus = 'paused';
     } else if (status.isGenerating) {
       playerStatus = 'generating';
     } else {
       playerStatus = 'idle';
     }

     // Update tracked state
     currentChunkIndex = status.currentChunkIndex ?? currentChunkIndex;
     currentTotalChunks = status.totalChunks ?? currentTotalChunks;
     currentSpeed = status.playbackSpeed ?? currentSpeed;

     // Update floating player if exists
     if (floatingPlayer) {
       floatingPlayer.update({
         status: playerStatus,
         currentChunk: currentChunkIndex,
         totalChunks: currentTotalChunks,
         playbackSpeed: currentSpeed
       });
     }

     return { success: true };
   }
   ```

4. **Request initial state on content script load:**
   At the end of the content script (after all handlers are set up), request current state:
   ```typescript

   chrome.runtime.sendMessage({
     target: 'service-worker',
     type: MessageType.GET_STATUS
   }).then((response) => {
     if (response.success) {
       handleStatusUpdate({ status: response });
     }
   }).catch(() => {
     // Service worker might not be ready
   });
   ```
